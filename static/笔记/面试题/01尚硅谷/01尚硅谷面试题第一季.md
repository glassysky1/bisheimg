## 作用域和值类型引用类型的传递

### 第一题 作用域

```js
    //第1题 作用域
    //全局的变量，会定义在window上
    var num1 = 55
    var num2 = 66
    function f1(num, num1) {
      //函数作用域
      //var num =55
      //var num1 =66
      //num 和num1都是私有的变量声明
      //num2没有声明，他指向的是全局的num2
      num = 100
      num1 = 100
      num2 = 100
      console.log(num)//100
      console.log(num1)//100
      console.log(num2)//100
    }
    //f1调用完就释放了
    // 55     66
    f1(num1, num2)

    console.log(num1)//55
    console.log(num2)//100
    console.log(num)//报错
```

### 第二题 值类型和引用类型的传递

```js
    function Person(name, age, salary) {
      this.name = name
      this.age = age
      this.salary = salary
    }
    function f1(person) {
      //var person = p，person指向了p指向的地址
      person.name = 'ls'
      person = new Person('aa', 18, 10)
    }

    var p = new Person('zs', 18, 1000)
    console.log(p.name)//zs
    f1(p)
    console.log(p.name)//'ls'

```
